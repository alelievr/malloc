struct alloc {
	void	*start;
	void	*end;
	struct alloc *next;
}

struct mem {
	int		page_type; //page allocation type: TINY, SMALL or LARGE
	int		free_bytes; //total number of free bytes in the block
	int		max_free_bytes_block; //maximum number of continous bytes in a free block
	
	void	*start;
	void	*end;
}
